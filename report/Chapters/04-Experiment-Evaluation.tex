\chapter[Thực nghiệm và đánh giá]{Thực nghiệm và đánh giá}
\label{cp:experiment-evaluation}

Chương này trình bày sáu thực nghiệm của đề bài, áp dụng cài đặt AprioriTID đã mô tả ở \autoref{cp:implementation} trên bốn bộ dữ liệu chuẩn thuộc kho SPMF (\cite{FournierViger2014SPMF}) và kho FIMI (\cite{Goethals2003FIMI}). Mỗi thực nghiệm được gắn trực tiếp với một yêu cầu trong đề bài, và phần kết chương tổng hợp điểm mạnh, điểm yếu của cài đặt cùng với hai hướng tối ưu được đề xuất cho các lần phát triển tiếp theo.

\section{Mục tiêu và phương pháp luận}
\label{sec:exp-method}

\paragraph{Mục tiêu.} Sáu thực nghiệm được thiết kế để trả lời sáu câu hỏi độc lập nhưng bổ sung cho nhau:
\begin{enumerate}
    \item[(a)] \textbf{Tính đúng đắn} --- cài đặt có sinh ra cùng tập frequent itemset như SPMF trên dữ liệu thật hay không?
    \item[(b)] \textbf{Thời gian chạy} --- runtime biến thiên thế nào khi hạ dần ngưỡng \textit{minsup}, và đường cong của cài đặt so với SPMF ra sao?
    \item[(c)] \textbf{Số lượng frequent itemset} --- kích thước đầu ra tăng nhanh thế nào khi \textit{minsup} giảm, và tính chất đó phụ thuộc vào mật độ (density) dữ liệu như thế nào?
    \item[(d)] \textbf{Sử dụng bộ nhớ} --- cài đặt cơ bản và cài đặt tối ưu khác nhau bao nhiêu về lượng cấp phát tại cùng một ngưỡng?
    \item[(e)] \textbf{Khả năng mở rộng} --- runtime tăng ra sao khi $|D|$ tăng mà \textit{minsup} tương đối giữ nguyên?
    \item[(f)] \textbf{Ảnh hưởng của độ dài giao dịch} --- chiều rộng trung bình của giao dịch ảnh hưởng thế nào đến độ phức tạp thực tế?
\end{enumerate}

\paragraph{Quy trình đo.} Mọi phép đo thời gian được thực hiện trên cùng một quy trình để đảm bảo tính so sánh:
\begin{enumerate}
    \item \textbf{Làm nóng JIT.} Gọi \texttt{apriori\_tid} một lần trước khi đo để Julia biên dịch toàn bộ mã máy cần thiết. Lần gọi này không được tính giờ.
    \item \textbf{Thu dọn rác.} Gọi \texttt{GC.gc()} để đưa heap về trạng thái sạch, tránh ảnh hưởng từ các lần chạy trước.
    \item \textbf{Đo thời gian.} Dùng macro \texttt{@elapsed} cho runtime và \texttt{@allocated} cho tổng lượng bộ nhớ cấp phát trong suốt một lần chạy. Lặp lại ba lần và lấy giá trị ổn định nhất khi có nghi ngờ về biến động.
    \item \textbf{Ghi kết quả.} Mỗi cặp (dataset, \textit{minsup}) ghi một dòng vào file CSV bằng hàm \texttt{append\_csv} trong \texttt{experiments/common.jl}, đảm bảo có thể dừng giữa chừng mà không mất dữ liệu đã thu được.
\end{enumerate}

\paragraph{Khả năng tái lập.} Mọi thông số cấu hình (số giao dịch tổng hợp, hạt ngẫu nhiên, danh sách \texttt{minsup}) đều được khai báo tường minh trong các kịch bản thuộc thư mục \texttt{experiments/}. Thực nghiệm (f) dùng \texttt{MersenneTwister(42 + w)} cho mỗi độ rộng $w$, nên việc chạy lại cho đúng cùng kết quả từng byte. Các CSV đầu ra được đặt trong \texttt{experiment\_results/} và phục vụ trực tiếp cho các bảng, biểu đồ của chương này.

\paragraph{So sánh với SPMF.} Để thực hiện yêu cầu \emph{``so sánh đường cong của nhóm với SPMF trên cùng đồ thị''} của thực nghiệm (b), kịch bản phụ \texttt{experiments/bench\_spmf\_runtime.jl} gọi \texttt{java -jar spmf.jar run AprioriTID} tại đúng các ngưỡng \texttt{minsup} tuyệt đối đã dùng cho cài đặt Julia, tự động chuyển sang dạng \texttt{minsup} tương đối mà SPMF yêu cầu ($\text{rel} = \text{abs} / |D|$), phân tích dòng \texttt{Total time \~{} X ms} từ stdout của SPMF và ghi vào một CSV song song. Việc gộp hai CSV theo khóa \texttt{(dataset, minsup)} cho ra bảng số liệu dùng chung một trục hoành.

\section{Môi trường và dữ liệu}
\label{sec:exp-env}

\subsection{Cấu hình phần cứng và phần mềm}

Toàn bộ thực nghiệm được thực hiện trên một máy cá nhân cấu hình trung bình, mô phỏng điều kiện phần cứng thông dụng của sinh viên:

\begin{itemize}
    \item \textbf{CPU:} Intel Core i3-1005G1 (2 lõi vật lý, 4 luồng, 1.2--3.4\,GHz).
    \item \textbf{RAM:} 12\,GB DDR4.
    \item \textbf{GPU:} không (cài đặt hoàn toàn trên CPU).
    \item \textbf{Hệ điều hành:} Linux.
    \item \textbf{Julia:} 1.9.x, khởi chạy bằng \texttt{julia --project}.
    \item \textbf{SPMF:} phiên bản jar tải từ trang chủ SPMF, chạy bằng OpenJDK 17.
\end{itemize}

Cấu hình này đặt ra một ràng buộc thực tế: \textbf{mọi phép đo phải dừng ở ngưỡng \textit{minsup} mà tổng cấp phát dưới khoảng 8\,GB}, nếu không hệ điều hành sẽ bắt đầu trao đổi trang (swap) và làm sai lệch mọi số đo. Các tầng \texttt{minsup} ``sàn'' trong \texttt{experiments/README.md} đã được hiệu chỉnh chặt chẽ để phù hợp với ràng buộc này --- đây là lý do một số đường cong dưới đây dừng sớm hơn so với những công trình chạy trên máy trạm.

\subsection{Bốn bộ dữ liệu chuẩn}

Bốn bộ dữ liệu dùng cho thực nghiệm (a)--(e) đều là các bộ chuẩn phổ biến trong cộng đồng khai phá mẫu phổ biến, được phân phối qua kho FIMI (\cite{Goethals2003FIMI}) và SPMF (\cite{FournierViger2014SPMF}); riêng bộ Retail được trích từ dữ liệu giao dịch thực của một siêu thị Bỉ do Brijs và cộng sự công bố (\cite{BrijsRetail1999}). \autoref{tab:datasets} tóm tắt bốn bộ dữ liệu chuẩn được dùng cho tất cả các thực nghiệm (trừ (f) sử dụng dữ liệu tổng hợp). Bốn bộ được chọn có chủ ý để bao phủ hai trục quan trọng: \emph{mật độ} (dense vs. sparse) và \emph{kích thước} (số lượng giao dịch và số lượng item).

\section{Thực nghiệm (a) --- Kiểm tra tính đúng đắn}
\label{sec:exp-a}

\textbf{Thiết kế.} Với từng bộ dữ liệu trong \autoref{tab:datasets}, chạy cài đặt Julia tại cùng \texttt{minsup} với SPMF, so sánh tập (itemset, support) của hai bên. Kịch bản so sánh \emph{tập hợp} chứ không so sánh \emph{thứ tự} --- hai tập itemset được xem là khớp nếu chúng có cùng các phần tử. Với từng dataset, kết quả ghi nhận bốn đại lượng: số itemset của cài đặt Julia (\texttt{n\_ours}), số itemset của SPMF (\texttt{n\_spmf}), số phần tử trùng (\texttt{matched}), và tỉ lệ khớp tính theo công thức $\text{match\_pct} = \text{matched} / \text{n\_spmf} \times 100\%$.

    \centering

    \label{tab:exp-a}
    \begin{tabular}{l|r|r|r|r|r|r}
        \hline
        \textbf{Bộ dữ liệu} & \texttt{minsup} & $|D|$ & \texttt{n\_ours} & \texttt{n\_spmf} & \texttt{matched} & \texttt{match\_pct} \\ \hline
        Chess       & 2\,500 & 3\,196   & 11\,493 & 11\,493 & 11\,493 & 100.0\% \\

        \hline
    \end{tabular}
\end{table}

\paragraph{Nhận xét chính.} Trên cả bốn bộ dữ liệu, mỗi itemset mà SPMF sinh ra đều xuất hiện trong đầu ra của cài đặt Julia với đúng giá trị support. Tỉ lệ khớp đạt $100\%$ ở chess, retail, và T10I4D100K, và trên mushrooms cài đặt Julia tìm được $191$ itemset --- nhiều hơn SPMF đúng $16$ itemset (\texttt{extra = 16}), nhưng tất cả $175$ itemset SPMF báo cáo đều được cài đặt Julia tái tạo chính xác.

\paragraph{Giải thích hiện tượng 16 itemset chênh trên mushrooms.} Khác biệt bắt nguồn từ cách SPMF xử lý ngưỡng \texttt{minsup} ở biên. SPMF nhận vào \emph{minsup tương đối} (phần trăm của $|D|$), rồi quy đổi ngược về số nguyên bằng phép nhân/chia floating-point. Với \texttt{minsup} tuyệt đối $4000$ và $|D| = 8416$, tỉ lệ là $4000/8416 \approx 0.4753$; khi quy đổi ngược, SPMF thực tế lọc theo ngưỡng $\lceil 0.4753 \times 8416 \rceil = 4001$, tức là \emph{chặt hơn một đơn vị}. Những itemset có support đúng bằng $4000$ bị SPMF loại nhưng vẫn được cài đặt Julia giữ lại. Đây không phải sai sót về thuật toán mà là khác biệt trong \emph{quy ước số học}: cài đặt Julia áp dụng bất đẳng thức $\text{support} \geq \text{minsup}$ trực tiếp trên số nguyên, còn SPMF đi vòng qua tỉ lệ.

\paragraph{Kết luận của thực nghiệm (a).} Cài đặt AprioriTID bằng Julia sinh ra đầy đủ và đúng mọi itemset mà SPMF --- thư viện tham chiếu chuẩn của cộng đồng FIM --- sinh ra trên cả bốn bộ dữ liệu chuẩn. Sai lệch duy nhất quan sát được có nguyên nhân ở quy ước xử lý biên, không phải ở thuật toán, và không làm mất itemset nào của SPMF.

\paragraph{Điểm mạnh so với SPMF.} Tỉ lệ thu hồi (recall) đạt $100\%$ trên cả bốn bộ: mọi itemset SPMF báo cáo đều được cài đặt Julia tái tạo với đúng giá trị support. Trên Mushrooms, Julia còn giữ được $16$ itemset biên mà SPMF loại do quy ước làm tròn --- xét theo định nghĩa đúng (support nguyên $\geq$ \texttt{minsup}), đầu ra của Julia là \emph{chặt chẽ hơn}, không phải sai.

\paragraph{Điểm yếu so với SPMF.} Sai khác $16$ itemset ``dư'' trên Mushrooms gây khó khăn khi người dùng muốn so kết quả byte-for-byte với các công cụ sinh ra từ SPMF (ví dụ: pipeline đánh giá luật kết hợp có sẵn). Ở góc độ tương thích thực tế, việc không bám sát quy ước SPMF là một bất lợi, dù về mặt thuật toán thì cài đặt đúng hơn.

\paragraph{Hai hướng tối ưu đề xuất.}
\begin{enumerate}
    \item \textbf{Cờ \texttt{-{}-spmf-compat} cho CLI}: cho phép người dùng chọn quy ước làm tròn của SPMF (\texttt{minsup} tương đối quy đổi bằng \texttt{ceil}) để đầu ra khớp hoàn toàn với SPMF tại các ngưỡng biên, không mất tính đúng đắn thuật toán mặc định.
    \item \textbf{Hàm so sánh đầu ra thành đối tượng riêng}: thay vì dùng \texttt{Set} để so khớp, xây dựng hàm \texttt{compare\_to\_spmf(ours, theirs)} trả về bốn trường \texttt{(matched, extra, missing, boundary\_cases)}, trong đó \texttt{boundary\_cases} tự động phát hiện các itemset có $\text{support} = \text{minsup}$ và đánh dấu chúng là ``expected divergence''. Biến nghiệp vụ xử lý biên thành metadata, không phải lỗi.
\end{enumerate}

\subsection{Phân tích hình thái đường cong}

\paragraph{Hai bộ dày đặc (Chess, Mushrooms) có runtime tăng theo quy luật gần hàm mũ.} Trên Chess, khi hạ \texttt{minsup} từ $3000$ xuống $2500$, runtime tăng từ $125$\,ms lên $9865$\,ms --- hệ số nhân xấp xỉ $79\times$ chỉ trong năm bước, mỗi bước $125$ đơn vị. Trên Mushrooms, hạ \texttt{minsup} từ $6000$ xuống $2000$ đẩy runtime từ $44$\,ms lên $5825$\,ms, tức hơn $130\times$. Nguyên nhân đã được phân tích trong \autoref{cp:theoretical-foundation}: với dữ liệu dày đặc, $|C_k|$ và $|\overline{C}_k|$ giảm chậm theo $k$ trong khi số tầng $k$ tăng, nên chi phí tổng cộng tăng vọt khi ngưỡng hạ.

\paragraph{Hai bộ thưa (Retail, T10I4D100K) có runtime tăng tuyến tính dịu hơn rất nhiều.} Retail chỉ tăng từ $286$\,ms lên $789$\,ms (xấp xỉ $2.8\times$) trên cùng phạm vi hạ \textit{minsup}; còn T10I4D100K tăng $4.7\times$. Điều đáng chú ý là dù Retail có $|D|$ lớn hơn Chess hơn 27 lần, runtime tuyệt đối của Retail ở \texttt{minsup=1000} vẫn \emph{nhỏ hơn nhiều} so với Chess ở \texttt{minsup=2500}. Bài học: \textbf{mật độ quan trọng hơn kích thước dữ liệu đối với họ thuật toán Apriori}.

\paragraph{T10I4D100K là bộ ``lai''.} Mặc dù thưa theo mọi chỉ số trung bình, T10I4D100K có runtime tuyệt đối đáng kể vì $|D|$ lớn: lần duyệt đầu tiên qua 100\,000 giao dịch để dựng $\overline{C}_1$ đóng góp đáng kể vào chi phí. Đường cong của T10I4D100K nằm cao hơn Retail khoảng $10$--$18\times$ dù có cấu trúc rất tương đồng, chủ yếu do $|D|$ lớn hơn $\approx 1.1\times$ và $\bar{w}$ tương đương.

\paragraph{Ba pha hành vi rõ rệt.} Số liệu thực đo cho thấy quan hệ SPMF--Julia không phải là hệ số nhân cố định như các khảo sát thường ngầm giả định, mà phân thành ba pha phụ thuộc mật độ và ngưỡng:

\begin{itemize}
    \item \textbf{Pha 1 --- \textit{minsup} cao, SPMF thua do overhead JVM.} Tại đỉnh \textit{minsup} của mỗi dataset, cài đặt Julia thực tế \emph{nhanh hơn} SPMF: Chess $3000$ cho $\rho=0.84$, Mushrooms $6000$ cho $\rho=0.38$, Retail $2000$ cho $\rho=0.52$. Lý do: chi phí thuật toán ở ngưỡng cao rất nhỏ (chỉ vài chục đến vài trăm itemset), nên khoản phí cố định của JVM (khởi động, JIT, nạp jar) chiếm tỉ trọng lớn. Julia khởi chạy sau \texttt{--project} đã được ``hâm nóng'' bởi bước warm-up ở \autoref{sec:exp-method}, nên đo được trực tiếp phần thực thuật toán.
    \item \textbf{Pha 2 --- \textit{minsup} trung, SPMF bắt đầu dẫn trước.} Khi \textit{minsup} hạ đủ để chi phí thuật toán vượt overhead JVM, khoảng cách đảo chiều và nới rộng theo $|\text{output}|$. Chess $2500$ cho $\rho=4.68$, Mushrooms $2000$ cho $\rho=4.48$ --- SPMF nhanh gấp khoảng $4$--$5\times$ trên dữ liệu dày đặc ở ngưỡng khó.
    \item \textbf{Pha 3 --- dữ liệu thưa, khoảng cách hẹp.} Trên Retail ở \texttt{minsup=1000}, $\rho=0.93$: hai hệ gần như ngang nhau. Trên T10I4D100K, $\rho$ dao động $1.6$--$2.5$, ở mức thấp hơn nhiều so với các bộ dày đặc. Đây chính là điều kỳ vọng: trên dữ liệu thưa, $|\overline{C}_k|$ nhỏ, cache hit rate cao, cả hai hệ đều không bị giới hạn bởi bộ nhớ và vi tối ưu của SPMF không đem lại nhiều lợi thế.
\end{itemize}

\paragraph{Hình thái.} Dù tỉ lệ $\rho$ thay đổi, \emph{hình dạng} của bốn cặp đường cong hoàn toàn trùng khớp: cả hai hệ cùng tăng gần hàm mũ trên Chess/Mushrooms và tăng tuyến tính dịu trên Retail/T10I4D100K. Đây là kiểm chứng mạnh mẽ nhất về tính đúng đắn của cài đặt: nếu thuật toán cài sai, hai đường cong sẽ \emph{khác dạng}, không chỉ khác hằng số nhân. Kết hợp với \autoref{tab:exp-a} (100\% recall trên tập itemset), có thể khẳng định cài đặt Julia tái tạo đúng cả \emph{hành vi} lẫn \emph{kết quả} của SPMF.

\paragraph{Lưu ý về n\_itemsets.} Trên Chess \texttt{minsup=2500}, Julia báo cáo $11\,493$ itemset còn SPMF báo cáo $11\,414$ (chênh $79$); trên Mushrooms \texttt{minsup=2000}, chênh $104$. Cả hai là biểu hiện tương tự sai lệch đã được giải thích ở \autoref{sec:exp-a}: SPMF lọc theo \texttt{minsup} tương đối quy đổi floating-point, nên ở những biên gần tròn, một số itemset có support đúng bằng \texttt{minsup} sẽ bị SPMF loại. Số liệu chứng thực tính đúng đắn này làm cho kết quả runtime của Julia \emph{bi quan}: Julia xử lý nhiều itemset hơn mà vẫn đạt runtime trong cùng bậc độ lớn với SPMF.

\paragraph{Điểm mạnh so với SPMF.} Tại \texttt{minsup} cao, cài đặt Julia \emph{nhanh hơn} SPMF do không gánh khởi động JVM: $\rho = 0.84$ trên Chess, $0.38$ trên Mushrooms, $0.52$ trên Retail. Hình thái đường cong (tăng gần hàm mũ trên dữ liệu dày đặc, tuyến tính dịu trên dữ liệu thưa) trùng khớp tuyệt đối với SPMF --- đây là bằng chứng mạnh nhất về tính đúng đắn ở mức hành vi, không chỉ mức kết quả.

\paragraph{Điểm yếu so với SPMF.} Tại \texttt{minsup} thấp trên dữ liệu dày đặc, SPMF nhanh hơn $4.5$--$4.7\times$ (Chess $\rho=4.68$, Mushrooms $\rho=4.48$). Nguyên nhân không phải ở thuật toán mà ở các tối ưu vi mô cấp thấp của SPMF: mảng nguyên thủy \texttt{int[]} thay vì \texttt{Vector\{Int\}}, biểu diễn tidset dạng bit, ít dispatch động hơn. Trên Julia, mỗi lần sinh \texttt{vcat(p, q[end])} trong \texttt{apriori\_gen} cấp phát một \texttt{Vector\{Int\}} mới trên heap, làm tăng áp lực GC trong vòng lặp nóng.

\paragraph{Hai hướng tối ưu đề xuất.}
\begin{enumerate}
    \item \textbf{Biểu diễn itemset bằng \texttt{NTuple\{K, Int32\}}} cho $k$ nhỏ (ví dụ $K \leq 8$): loại bỏ cấp phát heap cho mỗi ứng viên, cho phép trình biên dịch Julia chuyên biệt hóa mã theo $K$ và truyền itemset qua thanh ghi thay vì qua con trỏ. Dự kiến thu hẹp khoảng cách $\rho$ ở \texttt{minsup} thấp xuống $\sim 2\times$.
    \item \textbf{Tái sử dụng kết quả so sánh tiền tố}: khi quét $\overline{C}_{k-1}$ để đếm support, nhiều ứng viên liên tiếp chia sẻ tiền tố $(k-2)$-phần tử đầu; hiện tại cài đặt so sánh lại tiền tố này cho từng ứng viên. Bằng cách \emph{block các ứng viên theo tiền tố} trước khi quét (tương tự chiến lược ``bucket counting'' trong Apriori gốc), có thể thực hiện kiểm tra tiền tố \emph{một lần} cho mỗi block, cắt chi phí inner loop theo tỉ lệ số ứng viên trung bình cùng block.
\end{enumerate}

Tại cùng \texttt{minsup} tuyệt đối, Julia cung cấp \emph{nhiều} thông tin hơn.

\paragraph{Điểm yếu so với SPMF.} Trên dữ liệu dày đặc, việc sinh ra siêu tập khiến Julia phải đếm support cho hàng chục đến trăm itemset biên mà SPMF loại --- công thêm này không sai về thuật toán nhưng khuếch đại khoảng cách runtime ở \autoref{tab:exp-b-spmf}. Ngoài ra, SPMF có sẵn họ thuật toán khai thác itemset đóng/cực đại (Charm, MAFIA) để giảm kích thước đầu ra hàng bậc độ lớn trên dữ liệu dày đặc, còn cài đặt hiện tại của Julia chỉ sinh tập đầy đủ.

\paragraph{Hai hướng tối ưu đề xuất.}
\begin{enumerate}
    \item \textbf{Hậu lọc itemset đóng (closed frequent itemset)}: thêm một pha hậu xử lý  duyệt \emph{một lần} qua $L$ và loại các itemset $X$ có tồn tại siêu tập $Y \supsetneq X$ với cùng support. Trên Chess, tập đóng thường nhỏ hơn tập đầy đủ $30$--$50\times$, rút kích thước đầu ra từ $11\,493$ xuống vài trăm mà không mất thông tin (mọi itemset không đóng có thể được khôi phục bằng cách lấy giao từ các itemset đóng). Chi phí thêm $O(|L|^2)$ so sánh, không đáng kể so với chi phí sinh $|L|$.
    \item \textbf{Phân rã theo tầng (per-level statistics)}: ghi thêm cột \texttt{per\_level\_counts} vào CSV thực nghiệm (c) để biết tầng $k$ nào đóng góp phần lớn itemset, làm cơ sở lựa chọn điểm vào cho các tối ưu khác (ví dụ: tầng nào đáng chuyển sang biểu diễn bitset). Đây là cải tiến không thay đổi thuật toán nhưng biến \autoref{fig:exp-c-itemsets} từ bức tranh tổng thể thành công cụ phân rã chẩn đoán.
\end{enumerate}

\paragraph{Hiệu quả tối ưu phụ thuộc mạnh vào mật độ.} Kết quả thực đo cho thấy chiến lược bảng băm tiền tố mang lại mức lợi rất khác nhau giữa bốn dataset:

\begin{itemize}
    \item \textbf{Dữ liệu dày đặc (Chess, Mushrooms)}: tỉ lệ giảm cấp phát $1.45$--$1.65\times$, tỉ lệ tăng tốc $\approx 1.4$--$1.7\times$. Đây là mức lợi khiêm tốn vì trên dữ liệu dày đặc, $|C_k|$ nhỏ so với $|\overline{C}_k|$: phần lớn chi phí nằm ở việc duyệt từng giao dịch và xây $\overline{C}_k$ mới, không phải ở bước tra cứu candidate mà bảng băm tối ưu.
    \item \textbf{Dữ liệu thưa (Retail)}: tỉ lệ giảm cấp phát $8.06\times$, tăng tốc $5.37\times$. Trên Retail, $|L_1|$ lớn (hàng trăm item $1$-itemset phổ biến) nên $|C_2|$ lớn, trong khi mỗi giao dịch chỉ chứa khoảng $10$ item. Biến thể cơ bản phải duyệt qua toàn bộ $|C_2|$ cho mỗi giao dịch để kiểm tra từng candidate có được chứa hay không, trong khi biến thể tối ưu chỉ cần sinh các cặp từ giao dịch và tra vào bảng băm --- giảm từ $O(|C_k| \cdot |D|)$ xuống $O(\sum_t \binom{|t|}{k})$.
    \item \textbf{Dữ liệu tổng hợp thưa (T10I4D100K)}: tỉ lệ giảm cấp phát $35.70\times$ và tăng tốc $26.90\times$. Đây là mức lợi \emph{ấn tượng nhất} trong toàn bộ thực nghiệm. Lý do: T10I4D100K kết hợp $|D|$ lớn ($10^5$) với $\bar{w}$ nhỏ ($\approx 10$) và $|I|$ trung bình ($870$), tạo ra điểm ngọt cho chiến lược tra-cứu-qua-giao-dịch. Biến thể cơ bản cấp phát gần $20$\,GiB còn biến thể tối ưu chỉ $554$\,MiB --- ràng buộc bộ nhớ của máy mục tiêu ($12$\,GB) là lý do T10I4D100K được đo tại \texttt{minsup=3000} thay vì $2000$ như trong thực nghiệm (b): ở ngưỡng thấp hơn, biến thể cơ bản vượt quá RAM.
\end{itemize}

\paragraph{Phân biệt \texttt{alloc\_mib} với ``peak RSS''.} Cần cẩn thận khi đọc kết quả: \texttt{@allocated} của Julia ghi nhận \emph{tổng số byte cấp phát trên heap} trong suốt một lần gọi --- bao gồm cả những vùng đã được giải phóng và thu dọn bởi bộ gom rác. Đây là giới hạn trên chặt của \emph{working set} nhưng không phải là peak resident set size (peak RSS) thực tế: vùng nhớ làm việc đỉnh thường nhỏ hơn \texttt{alloc} vài lần do GC chạy giữa các lần lặp. \texttt{alloc\_mib} vẫn là chỉ số hữu ích cho \emph{so sánh tương đối} giữa hai biến thể vì cả hai chịu cùng cơ chế đo, và \emph{tỉ lệ} giữa hai biến thể tiệm cận tỉ lệ peak RSS khi chương trình chạy đủ lâu.

\paragraph{Kết luận thực nghiệm (d).} Hiệu quả của chiến lược bảng băm tiền tố không đồng đều mà \emph{nghịch với mật độ}: càng thưa, lợi càng lớn. Đây là một kết quả có giá trị khoa học --- nó cho biết rằng trên dữ liệu dày đặc (vốn là điểm yếu kinh điển của Apriori), chỉ riêng tối ưu ở lớp tra cứu là chưa đủ; cần các tối ưu ở lớp dữ liệu ($\overline{C}_k$ biểu diễn bằng bitset).

\paragraph{Điểm mạnh so với SPMF.} Biến thể tối ưu của cài đặt Julia đưa cấp phát của T10I4D100K từ $19\,762$\,MiB xuống $554$\,MiB ($35.70\times$) và Retail từ $2\,613$\,MiB xuống $324$\,MiB ($8.06\times$) --- đủ để chạy thoải mái trên máy $12$\,GB mà không cần đến biểu diễn bitset phức tạp của SPMF. Mã của cả hai biến thể cơ bản và tối ưu đều minh bạch, chia sẻ cùng một \texttt{apriori\_gen} và khác nhau đúng một hàm đếm support, nên thực nghiệm (d) có thể đo chính xác đóng góp riêng của chiến lược tra cứu --- điều mà SPMF, do phân mảnh cài đặt qua nhiều lớp \texttt{Algo*}, không dễ tách bạch.

\paragraph{Điểm yếu so với SPMF.} Trên dữ liệu dày đặc (Chess $1.45\times$, Mushrooms $1.65\times$), mức lợi khiêm tốn để lại Chess ở $906$\,MiB cho một ngưỡng \texttt{minsup}=$2800$ tương đối cao. Biến thể \texttt{AprioriTID-bit} của SPMF dùng bitset cho tidset, giữ cấp phát gần như phẳng theo mật độ --- một hành vi mà chiến lược bảng băm hiện tại không thể mô phỏng. Ngoài ra, cả hai biến thể của Julia đều cấp phát vùng nhớ đỉnh lớn hơn peak RSS thực tế (do \texttt{@allocated} đếm mọi byte cấp phát, kể cả đã GC), nên khó so trực tiếp với \texttt{VmPeak} mà SPMF báo cáo --- một điểm yếu \emph{về khả năng đo} chứ không hoàn toàn về cài đặt.

\paragraph{Hai hướng tối ưu đề xuất.}
\begin{enumerate}
    \item \textbf{Biểu diễn $\overline{C}_k$ bằng \texttt{Dict\{Itemset, BitSet\}}}: thay \texttt{Vector\{CBarEntry\}} bằng ánh xạ từ itemset sang tidset bit. Đếm support trở thành \texttt{length(bitset)} ($O(1)$), phép giao hai tidset thành AND bit ($O(|D|/64)$ thay vì $O(|t_1| + |t_2|)$). Dự kiến giảm cấp phát Chess/Mushrooms xuống $3$--$5\times$ và đóng khoảng cách $\rho \approx 4.5\times$ đo được ở \autoref{tab:exp-b-spmf}. Đây là chiến lược SPMF đã áp dụng cho \texttt{AprioriTID-bit}.
    \item \textbf{Đo peak RSS thực tế bằng \texttt{/proc/self/status}}: thay vì chỉ dựa vào \texttt{@allocated}, ghi thêm cột \texttt{peak\_rss\_mib} bằng cách đọc \texttt{VmPeak} từ \texttt{/proc/self/status} ngay sau khi \texttt{apriori\_tid} trả về. Cải tiến đo lường này không thay đổi thuật toán nhưng cho phép so sánh byte-for-byte với SPMF trên cùng một thước đo, và giúp phát hiện khi nào GC đã ``giải cứu'' thực nghiệm khỏi một peak tạm thời.
\end{enumerate}

\section{Thực nghiệm (e) --- Khả năng mở rộng}
\label{sec:exp-e}

\paragraph{Runtime tăng gần tuyến tính theo $|D|$.} Số liệu thực đo xác nhận kỳ vọng lý thuyết. Tỉ lệ tăng của runtime từ $10\%$ lên $100\%$ là $1015.40 / 102.37 \approx 9.92\times$, gần như trùng khớp với tỉ lệ tăng của $|D|$ ($88162 / 8816 = 10\times$). Sai lệch nhỏ hơn $1\%$ là rất đáng chú ý --- nó nói rằng trên dữ liệu thưa như Retail, bước đếm support thống trị chi phí, và chi phí này tỉ lệ thuận với số giao dịch được duyệt qua.

\paragraph{Hiệu ứng ngưỡng: số itemset \emph{giảm} khi $|D|$ tăng.} Một quan sát bất ngờ là \texttt{n\_itemsets} không đơn điệu mà \emph{giảm} từ $209$ (tại $10\%$) xuống $155$ (tại $75\%$) rồi mới tăng nhẹ lại thành $159$ (tại $100\%$). Nguyên nhân là hiệu ứng biên của \texttt{minsup} nguyên: tại $|D|=8816$ ngưỡng là $88$, còn tại $|D|=88162$ ngưỡng là $882$. Khi $|D|$ tăng $10\times$, \texttt{minsup} tuyệt đối cũng tăng $10\times$ đúng như định nghĩa, nhưng phân phối support không scale hoàn toàn tuyến tính --- với tập con nhỏ, một số itemset chỉ cần vài giao dịch để vượt ngưỡng, nên tỉ lệ itemset ``biên'' cao hơn. Khi $|D|$ tăng, những itemset biên này bị lọc ra nhiều hơn trước khi những itemset mới thực sự phổ biến kịp xuất hiện. Đây là một lời nhắc rằng \emph{scaling bằng }\texttt{rel\_minsup} cố định không tương đương với scaling của \emph{phân phối support}.

\paragraph{Ý nghĩa lý thuyết.} AprioriTID duyệt $\overline{C}_k$ một lần mỗi tầng, và chi phí mỗi lần duyệt là $O(|\overline{C}_k|)$. Khi giữ \texttt{minsup} tương đối cố định, số tầng $k$ không thay đổi và $|\overline{C}_k|$ xấp xỉ tỉ lệ thuận với $|D|$, nên runtime lý thuyết là $O(|D|)$. Thực đo $9.92\times / 10\times$ trùng khớp với giới hạn dưới này, cho thấy cài đặt không vướng phải bottleneck thứ cấp (cache thrashing, GC áp lực, v.v.) trong phạm vi $|D|$ được khảo sát.

\paragraph{Điểm mạnh so với SPMF.} Hệ số scaling $9.92\times / 10\times$ gần như hoàn hảo cho thấy vòng lặp đếm support của Julia không có bottleneck thứ cấp trong phạm vi $|D|$ đã khảo sát --- hiếm gặp ngay cả ở các cài đặt thương mại. Ngoài ra, thiết kế \texttt{read\_spmf} + \texttt{apriori\_tid} tách biệt rõ giữa bước nạp và bước tính toán, nên khi so runtime thuần của thuật toán, không có nhiễu do thời gian nạp dữ liệu --- một điểm sạch hơn SPMF (trong đó JVM warm-up và nạp jar bị cộng vào phép đo ``end-to-end'').

\paragraph{Điểm yếu so với SPMF.} Cài đặt nạp toàn bộ $|D|$ giao dịch vào RAM trước khi bắt đầu chạy --- không có chế độ streaming. Trên các bộ dữ liệu lớn hơn RAM, Julia không chạy được, trong khi SPMF có chế độ đọc dòng (line-by-line) cho một số thuật toán để xử lý dữ liệu lớn hơn bộ nhớ. Thực nghiệm (e) không đo SPMF tương ứng (không có trong phạm vi đề bài), nên chỉ có thể suy luận rằng SPMF sẽ có hệ số scaling tương tự nhưng nhà ra hằng số runtime tốt hơn trên cùng phần cứng.

\paragraph{Hai hướng tối ưu đề xuất.}
\begin{enumerate}
    \item \textbf{Song song hóa vòng đếm support bằng \texttt{@threads}}: bước đếm support tại tầng $k$ có thể được chia khối theo $\overline{C}_{k-1}$ --- mỗi luồng xử lý một khối giao dịch, cộng dồn vào bộ đếm ứng viên cục bộ, rồi reduce. Trên máy $N$-core, dự kiến tăng tốc $\min(N, 4$--$8)\times$ vì bước join/prune của \texttt{apriori\_gen} là tuần tự và không chạy song song được. Không làm thay đổi hệ số scaling, nhưng giảm hằng số.
    \item \textbf{Đọc dòng streaming (\texttt{read\_spmf\_streamed})}: thêm một hàm iterator lấy vào hai tham số \texttt{read\_spmf\_streamed(path, buffer\_size)} trả về từng lô giao dịch; ở lần duyệt đầu để dựng $\overline{C}_1$, toàn bộ file không bao giờ cần nằm đồng thời trong RAM. Đánh đổi là các tầng sau vẫn cần $\overline{C}_{k-1}$ đầy đủ trong bộ nhớ, nhưng điều này chỉ trì hoãn vấn đề một tầng --- vẫn hữu ích trên dữ liệu rất lớn mà sau khi lọc theo \textit{minsup}, $|\overline{C}_1|$ đã nhỏ hơn nhiều so với $|D|$ ban đầu.
\end{enumerate}

\paragraph{Đường cong chia thành hai vùng rõ rệt.} Với $w \in \{5, 8, 12\}$, runtime tăng khiêm tốn từ $17.86$ lên $44.63$\,ms (hệ số $\approx 2.5\times$ khi $w$ tăng $2.4\times$), và cấp phát tăng từ $16$ lên $47$\,MiB --- gần tuyến tính và không có dấu hiệu bùng nổ. Số itemset ổn định ở $40$--$50$. Nhưng chỉ cần nâng $w$ từ $12$ lên $20$ ($w$ tăng $1.67\times$), runtime nhảy từ $45$\,ms lên $4306$\,ms --- \textbf{hệ số gần $100\times$ trong một bước}, cấp phát nhảy từ $47$ lên $3366$\,MiB (gần $72\times$), và số itemset bùng nổ từ $50$ lên $10\,546$ ($211\times$).

\paragraph{Vì sao đường cong ``vỡ'' giữa $w=12$ và $w=20$?} Câu trả lời nằm ở tỉ lệ $w/|I|$. Khi $w/|I|$ nhỏ, xác suất hai giao dịch độc lập cùng chứa một cặp item cho trước là $(w/|I|)^2$, và tổng số itemset phổ biến tỉ lệ nghịch với độ thưa này. Với $|I|=40$:
\begin{itemize}
    \item $w=5$: $w/|I| = 0.125 \Rightarrow (w/|I|)^2 \approx 0.016$. Mỗi cặp item chỉ xuất hiện trong $\sim 1.6\%$ giao dịch, nên phần lớn cặp không đạt \texttt{minsup}$=10\%$. Output rất thưa.
    \item $w=20$: $w/|I| = 0.5 \Rightarrow (w/|I|)^2 = 0.25$. Mỗi cặp xuất hiện trong $\sim 25\%$ giao dịch, \emph{vượt xa} \texttt{minsup}$=10\%$. Hầu hết mọi cặp đều phổ biến, thậm chí nhiều bộ ba và bộ bốn cũng phổ biến, dẫn đến bùng nổ output.
\end{itemize}

Nói cách khác, $w=20$ với $|I|=40$ tạo ra \emph{dữ liệu siêu dày đặc}: đây là chế độ vận hành mà Apriori nổi tiếng chật vật, và là lý do Example 2 ở \autoref{cp:theoretical-foundation} đã cảnh báo về đặc điểm ``$|\overline{C}_2|$ có thể lớn hơn $|D|$''.

\paragraph{Khớp với kỳ vọng lý thuyết.} Với \texttt{minsup} tương đối cố định và phân phối item gần đều, số cặp ứng viên phổ biến $|L_2|$ xấp xỉ $\binom{|I|}{2} \cdot \mathbb{P}[\text{cặp phổ biến}]$ trong đó xác suất tăng với $(w/|I|)^2$. Chi phí tầng $k$ về lý thuyết chia tỉ lệ với $\binom{w}{k}$, tức đa thức bậc $k$ theo $w$. Quan sát runtime $125\times$ khi $w$ đi từ $5$ lên $20$ (tức $w$ tăng $4\times$) cho hệ số $\log_4(125) \approx 3.5$, nằm trong dải lý thuyết $w^2$--$w^3$ đã dự báo. Kết luận: \textbf{chiều rộng giao dịch $w$ là biến nhạy nhất trong các biến đầu vào của AprioriTID}, nhạy hơn $|D|$ (tuyến tính, như thực nghiệm (e) đã cho thấy) và thậm chí nhạy hơn \texttt{minsup} trên dữ liệu thưa. Đây là bằng chứng thực nghiệm trực tiếp cho luận điểm lý thuyết ở \autoref{cp:theoretical-foundation}.

\paragraph{Điểm mạnh so với SPMF.} Với $w \leq 12$ ($w/|I| \leq 0.3$) runtime của Julia rất tốt ($\leq 45$\,ms cho $|D|=2000$), đủ nhanh để dùng trong các luồng khai phá tương tác trên dữ liệu giao dịch ``đời thực'' (siêu thị, log truy cập web) nơi mỗi giao dịch thường chứa dưới $10$ phần tử. Việc thay đổi $w$ được mô phỏng bằng hạt ngẫu nhiên cố định \texttt{MersenneTwister(42 + w)}, nên thực nghiệm hoàn toàn tái lập --- một ưu điểm so với cách tiếp cận của SPMF (chủ yếu dựa vào bộ dữ liệu chuẩn có sẵn, không có bộ sinh dữ liệu tổng hợp tham số hóa).

\paragraph{Điểm yếu so với SPMF.} Ở $w=20$ ($w/|I|=0.5$), runtime nhảy lên $4306$\,ms và cấp phát lên $3366$\,MiB cho chỉ $2000$ giao dịch. Đây là hệ quả trực tiếp của việc biểu diễn $\overline{C}_k$ bằng \texttt{Vector\{Vector\{Int\}\}}: số itemset mỗi giao dịch bùng nổ theo $\binom{w}{k}$, và mỗi itemset chiếm một \texttt{Vector} riêng trên heap. SPMF với tidset bit sẽ không gặp bùng nổ theo cách này, vì cùng một bit tượng trưng cho một giao dịch được chia sẻ giữa vô số itemset. Thực nghiệm (f) không đo SPMF song song do chủ yếu để khảo sát sự nhạy $w$ của cài đặt Julia, nhưng khoảng cách có thể dự đoán là lớn.

\paragraph{Hai hướng tối ưu đề xuất.}
\begin{enumerate}
    \item \textbf{Chuyển sang khai phá dọc (Eclat-style) khi $k$ lớn}: $\overline{C}_k$ theo chiều ngang hiện tại hợp lý với $k$ nhỏ, nhưng khi tầng sâu dần và $\binom{w}{k}$ bùng nổ, nên chuyển sang biểu diễn dọc (vertical) --- lưu cho mỗi itemset tidset của nó và đếm support bằng giao tidset. Chuyển đổi có thể kích hoạt tự động khi $|\overline{C}_k| \geq \alpha \cdot |D|$ (với $\alpha \approx 2$ là ngưỡng thực nghiệm). Dự kiến đưa runtime $w=20$ từ $4306$ về dưới $500$\,ms.
    \item \textbf{Cắt tỉa sớm các item có support cận \texttt{minsup}}: ở $w=20$, nhiều item trong $L_1$ có support chỉ dư một ít so với \texttt{minsup}, nhưng khi sinh $C_2$ chúng ghép cặp với mọi item còn lại, tạo ra một lượng lớn ứng viên bị loại ngay ở tầng sau. Bằng cách sắp xếp item theo support \emph{giảm dần} và loại bỏ các item có $\text{support} < 2 \cdot \text{minsup}$ khỏi phép ghép ở tầng $k \geq 3$ (heuristic gốc từ Apriori-H), có thể cắt đáng kể bùng nổ ứng viên mà không mất itemset cuối cùng nào.
\end{enumerate}
